template <typename A>
string to_string(A v) {
  bool first = true;
  string res = "{";
  for (const auto& x : v) {
    if (!first) {
      res += ", ";
    }
    first = false;
    res += to_string(x);
  }
  res += "}";
  return res;
}

ll n, cnt, ans, a[100], b[100], c, d, tmp, m, h, w, x, y, sum, k, q;
ll dp[ohara][20];

int main(void) {
  cin.tie(0);
  cout.tie(0);
  ios::sync_with_stdio(false);

  cin >> n;
  rep(i, n) cin >> a[i];
  rep(i, n) cin >> b[i];

  // dp[mask][i]
  // maskで1のやつを既に並べてある時の最小
  //を知りたかったが、cost計算の際に、
  //最後にaddしたindexである所のi必要なので添え字に組み込む

  rep(i, ohara) rep(j, 20) dp[i][j] = LINF;
  rep(i, n) { dp[(1LL << i)][i] = 0; }
  ll lim = (1LL << n);
  rrep(i, 1, lim) {
    rep(j, n) {
      if (!(i & (1LL << j))) continue;
      rep(k, n) {
        if (i & (1LL << k)) continue;
        // sorted判定
        ll tatu = __builtin_popcount(i);
        ll before, after, dis, cost = 0;

        dis = abs(j - (tatu - 1));
        if (dis % 2 == 0) {
          //表
          before = a[j];
        } else {
          before = b[j];
        }

        dis = abs(k - tatu);
        if (dis % 2 == 0) {
          //表
          after = a[k];
        } else {
          after = b[k];
        }

        if (before > after) continue;

        rrep(z, k + 1, n) {
          if (i & (1LL << z)) {
            cost++;
          }
        }

        dp[(i | (1LL << k))][k] = min(dp[(i | (1LL << k))][k], dp[i][j] + cost);
      }
    }
  }

  ans = LINF;
  rep(i, 20) { ans = min(ans, dp[lim - 1][i]); }
  if (ans == LINF) ans = -1;
  Cout(ans);
  return 0;
}
