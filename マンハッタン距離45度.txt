template <typename A>
string to_string(A v) {
  bool first = true;
  string res = "{";
  for (const auto& x : v) {
    if (!first) {
      res += ", ";
    }
    first = false;
    res += to_string(x);
  }
  res += "}";
  return res;
}

int main(void) {
  cin.tie(0);

  ios::sync_with_stdio(false);

  cin >> h >> w >> d;
  //(x,y) → (x+y,x-y)に置換
  //戻す時はansy=(i+j)/2,ansx=i-ansy;
  //その際にx-yってのは、負の数にもなりうるので、(W-1)だけ平行移動させる(105行目のj-(W-1)のところ)

  rep(i, 1200) {
    rep(j, 1200) {
      stx = d * j, sty = d * i;
      if (stx > 1100 || sty > 1100) continue;
      rrep(zx, stx, stx + d) {
        if (zx > 1100) break;
        rrep(zy, sty, sty + d) {
          if (zy > 1100) break;
          if (i % 2 == 0) {
            // G or B
            if (j % 2 == 0)
              now_ans[zy][zx] = 'G';
            else
              now_ans[zy][zx] = 'B';
          } else {
            if (j % 2 == 0)
              now_ans[zy][zx] = 'Y';
            else
              now_ans[zy][zx] = 'R';
          }
        }
      }
    }
  }

  rep(i, 1100) {
    rep(j, 1100) {
      ll ansx, ansy;
      if ((i + j - (w - 1)) % 2 == 1) continue;
      ansy = (i + j - (w - 1)) / 2;
      ansx = i - ansy;
      if (ansx < 0 || ansx >= w || ansy < 0 || ansy >= h) continue;
      ans[ansy][ansx] = now_ans[i][j];
    }
  }
  rep(i, h) {
    rep(j, w) { cout << ans[i][j]; }
    cout << "\n";
  }
  return 0;
