void dfs(ll dokomade,ll yen,ll nannko){
    if(nannko>k)return;
    if(nannko==k){
        cnt=0;
        tmp=yen;
        while(1){
            if(tmp==0)break;
            rrrep(i,hiku_cou-1,0){
                if(tmp>=hiku[i]){
                    tmp-=hiku[i];
                    cnt++;
                    break;
                }
            }
        }
        ans=min(ans,cnt);
    }
    if(dokomade==n)return;
    dfs(dokomade+1,yen+a[dokomade],nannko+1);
    dfs(dokomade+1,yen,nannko);
}
