int main(void)
{
    cin.tie(0);
    ios::sync_with_stdio(false);

    cin >> n >> a >> b >> c;
    rep(i, n) cin >> l[i];
    for (int bit = 0; bit < (1 << 2 * n); bit++)
    {
        ll _a = 0, _b = 0, _c = 0, syn = 0;
        rep(i, n)
        {
            if (((bit >> (2 * i)) & 3) == 0)
                continue;
            if (((bit >> (2 * i)) & 3) == 1)
            {
                if (_a != 0)syn++;
                _a += l[i];
            }
            if (((bit >> (2 * i)) & 3) == 2)
            {
                if (_b != 0)syn++;
                _b += l[i];
            }
            if (((bit >> (2 * i)) & 3) == 3)
            {
                if (_c != 0)syn++;
                _c += l[i];
            }
        }
        if (_a == 0 || _b == 0 || _c == 0)continue;
        cnt = abs(a - _a) + abs(b - _b) + abs(c - _c) + syn * 10;
        ans = min(ans, cnt);
    }
    Cout(ans);
    return 0;
}
