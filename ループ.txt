template <typename A>
string to_string(A v) {
  bool first = true;
  string res = "{";
  for (const auto& x : v) {
    if (!first) {
      res += ", ";
    }
    first = false;
    res += to_string(x);
  }
  res += "}";
  return res;
}

U_MAP mp;

int main(void) {
  cin.tie(0);
  cout.tie(0);
  ios::sync_with_stdio(false);

  cin >> n >> x >> m;
  ll now = x, len = 0, ki;
  rep(i, n) {
    ans += now;
    mp[now]++;
    now = (now * now) % m;
    // Cout(ans);
    if (mp[now] == 2) {  //ループの頭発見
      ki = now;
      while (1) {
        if (now == ki && len != 0)
          break;  //頭から一周して見つけたらbreak(最初でbreakはつまらんのでlen!=0
        sum += now;
        now = (now * now) % m;
        len++;
      }
      //一旦一周,len=周期の長さ,nowは値
      ll now_ans = ((n - 1 - i) / len) * sum;
      ll now_k = ((n - 1 - i) % len);
      //残りの長さをlenでわって、スコアを*sum(一周のコスト)でO(1)で省略
      // now_kはあまりをシミュレーションするための余りの長さ
      now = ki;
      while (1) {
        if (now_k == 0) break;
        ans += now;
        now = (now * now) % m;
        now_k--;
      }
      //余りのシミュレーション
      ans += now_ans;
      break;
    }
  }
  Cout(ans);
  return 0;
}
