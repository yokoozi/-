template <typename A>
string to_string(A v) {
  bool first = true;
  string res = "{";
  for (const auto& x : v) {
    if (!first) {
      res += ", ";
    }
    first = false;
    res += to_string(x);
  }
  res += "}";
  return res;
}

int main(void) {
  cin.tie(0);

  cin >> n;
  rep(i, n) cin >> a[i];

  rep(i, n + 1) rep(j, n + 1) dp[i][j] = LINF;

  rep(i, n) {
    dp[i][i] = 1;
    val[i][i] = a[i];
  }
  //区間DPは[i:j]のj-iつまり、indexの差を1から〜小さい順に固定してdp[i][j]を変更する
  //区間でぷよぷよの類（連鎖反応）とかはok
  //val[i:j]はa[i:j]の値
  rrep(diff, 1, n) {
    rep(i, n) {
      ll j = i + diff;
      if (j >= n) continue;
      rrep(k, i, j) {
        if (dp[i][k] == 1 && dp[k + 1][j] == 1 && val[i][k] == val[k + 1][j]) {
          dp[i][j] = 1;
          val[i][j] = val[i][k] + 1;
        } else {
          dp[i][j] = min(dp[i][j], dp[i][k] + dp[k + 1][j]);
        }
      }
    }
  }
  Cout(dp[0][n - 1]);
  return 0;
}
