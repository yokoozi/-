void dijkstra(ll s){
        priority_queue<pair<ll,ll>,vector<pair<ll,ll>>,greater<pair<ll,ll>>>que;
        fill(d,d+n,LINF);
        d[s]=0;
        que.push(pair<ll,ll>(0LL,s));
        while(!que.empty()){
            pair<ll,ll> p=que.top();que.pop();
            ll v=p.second;
            if(d[v]<p.first)continue;
            rep(i,Size(g[v])){
                edge e=g[v][i];
                if(d[e.to]>d[v]+e.cost){
                    d[e.to]=d[v]+e.cost;
                    que.push(pair<ll,ll>(d[e.to],e.to));
                }
            }
        }
}
