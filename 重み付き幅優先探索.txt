ll bfs(ll mi){
        queue<pair<int,int>> que;
        rep(i,n){
            rep(j,m){
                d[i][j]=10000000000;
            }
        }
        que.push(pair<int,int>(sy,sx));
        d[sy][sx]=0;
        while(Size(que)){
            pair<int,int> p=que.front();que.pop();
            rep(i,4){
                int nx=dx[i]+p.second,ny=dy[i]+p.first;
                if(nx>=0&&nx<m&&ny>=0&&ny<n){
                    ll tmp=d[p.first][p.second];
                    if(s[ny][nx]=='#'){
                        tmp+=mi;
                    }
                    else{
                        tmp++;
                    }
                    if(tmp<d[ny][nx]){
                        que.push(pair<int,int>(ny,nx));
                        d[ny][nx]=tmp;
                    }
                }
            }
        }
        return d[gy][gx];
}

int main(void){
       cin.tie(0);
    ios::sync_with_stdio(false);

       cin >> n >> m >> t;
       rep(i,n){
           rep(j,m){
               cin>>s[i][j];
               if(s[i][j]=='S'){
                   sx=j;
                   sy=i;
               }
               else if(s[i][j]=='G'){
                   gx=j;
                   gy=i;
               }
           }
       }
       ll  le=0,ri=2000000008,mi;
       while(ri-le>1){
           mi=(le+ri)/2;
           pos=bfs(mi);
           if(pos<=t){
               le=mi;
           }
           else{
               ri=mi;
           }
       }
       Cout(le);
return 0;
}
