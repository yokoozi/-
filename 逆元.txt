ll modinv(ll aa,ll mm){
    ll bb=mm,uu=1,vv=0;
    while(bb){
        ll tt=aa/bb;
        aa-=tt*bb;swap(aa,bb);
        uu-=tt*vv;swap(uu,vv);
    }
    uu%=mm;
    if(uu<0)uu+=mm;
    return uu;
}
