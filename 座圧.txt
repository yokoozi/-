U_MAP fukugen;
template <class T>
vector<ll> Zarts(const vector<T> &AR, ll CNT = 0) {
  vector<ll> RES;
  map<T, ll> MP;
  for (auto &e : AR) MP[e] = 0;
  for (auto &e : MP) e.se = CNT++;
  for (auto &e : AR) {
    fukugen[MP[e]] = e;
    RES.push_back(MP[e]);
  }
  return RES;
}

  return 0;
}
