ll sum(ll i){
  ll s=0;
  while(i>0){
    s+=bit[i];
    i-=i&-i;
  }
  return s;
}

void add(ll i,ll x){
  while(i<=ohara-10){
    bit[i]+=x;
    i+=i&-i;
  }
}
